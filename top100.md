### [128. Longest Consecutive Sequence](https://leetcode.com/problems/longest-consecutive-sequence/) (25/4)

```typescript
function longestConsecutive(nums: number[]): number {
  if (!nums.length) return 0;

  let longest = 1;
  const set = new Set(nums);

  for (const num of set) {
    if (set.has(num - 1)) continue;
    let length = 1;
    while (set.has(num + length)) {
      length++;
    }
    longest = Math.max(longest, length);
  }
  return longest;
}


// Time: O(n)
// Space: O(n)
```

### ***[560. Subarray Sum Equals K](https://leetcode.com/problems/subarray-sum-equals-k/) (26/4)

```typescript
function subarraySum(nums: number[], k: number): number {
  // sum(i, j) = prefixSum[j] - prefixSum[i-1]
  // prefixSum[j] - prefixSum[i-1] = k
  // prefixSum[i-1] = prefixSum[j] - k

  let count = 0;
  let prefixCount = new Map<number, number>();
  prefixCount.set(0, 1);
  let prefixSum = 0;

  for (const num of nums) {
    prefixSum += num;
    const need = prefixSum - k;

    if (prefixCount.has(need)) {
      count += prefixCount.get(need)!;
    }

    prefixCount.set(prefixSum, (prefixCount.get(prefixSum) ?? 0) + 1);
  }

  return count;
}

// Time: O(n)
// Space: O(n)

```



### ** [347. Top K Frequent Elements](https://leetcode.com/problems/top-k-frequent-elements/) (2/5)

```typescript
function topKFrequent(nums: number[], k: number): number[] {

    const freq = new Map<number, number>();

    for (const num of nums) {
        freq.set(num, (freq.get(num) ?? 0) + 1);
    }

    const pq = new MaxPriorityQueue({
        compare: (a, b) => {
            if (a[0] !== b[0]) return b[0] - a[0];
            return b[1] - a[1];
        }
    })

    for (const [num, count] of freq) {
        pq.enqueue([count, num]);
    }

    const res: number[] = [];

    for (let i = 0; i < k; i++) {
        res.push(pq.dequeue()[1]);
    }

    return res;
};

// Time: O(nlog(n))
// Space: O(n)



function topKFrequent(nums: number[], k: number): number[] {
  const freq = new Map<number, number>();

  for (const n of nums) {
    freq.set(n, (freq.get(n) ?? 0) + 1);
  }

  const bucket = Array.from({ length: nums.length + 1 }, () => []);

  for (const [num, count] of freq) {
    bucket[count].push(num);
  }

  const res: number[] = [];

  // scan n+1 times in the worst case
  for (let i = bucket.length - 1; i >= 0 && res.length < k; i--) {
    for (const n of bucket[i]) {
      res.push(n);
      if (res.length === k) break;
    }
  }

  return res;
}

// Time: O(n) 
// Space: O(n)
```

### **[49. Group Anagrams](https://leetcode.com/problems/group-anagrams/) (2/5)

```typescript
function groupAnagrams(strs: string[]): string[][] {
  const freq = new Map<string, string[]>();

  for (const str of strs) {
    const sorted = str.split('').sort().join('');

    if (!freq.has(sorted)) {
      freq.set(sorted, []);
    }

    freq.get(sorted).push(str);
  }

  const res: string[][] = [];

  for (const [key, value] of freq) {
    res.push([...value]);
  }

  return res;
}


// Time: O(n * mlog(m))
// Space: O(m*n)

function groupAnagrams(strs: string[]): string[][] {
  const freq = new Map<string, string[]>();

  for (const str of strs) {
    const bucket = new Array(26).fill(0);

    for (let ch of str) {
      bucket[ch.charCodeAt(0) - 'a'.charCodeAt(0)]++;
    }

    const key = bucket.join('#');

    if (!freq.has(key)) {
      freq.set(key, []);
    }

    freq.get(key).push(str);
  }

  const res: string[][] = [];

  for (const [_, value] of freq) {
    res.push([...value]);
  }

  return res;
}

// Time: O(m*n)
// Space: O(m*n)
```
