```typescript
function longestConsecutive(nums: number[]): number {
  if (!nums.length) return 0;

  let longest = 1;
  const set = new Set(nums);

  for (const num of set) {
    if (set.has(num - 1)) continue;
    let length = 1;
    while (set.has(num + length)) {
      length++;
    }
    longest = Math.max(longest, length);
  }
  return longest;
}


// Time: O(n)
// Space: O(n)
```

### ***[560. Subarray Sum Equals K](https://leetcode.com/problems/subarray-sum-equals-k/) (26/4)

```typescript
function subarraySum(nums: number[], k: number): number {
  // sum(i, j) = prefixSum[j] - prefixSum[i-1]
  // prefixSum[j] - prefixSum[i-1] = k
  // prefixSum[i-1] = prefixSum[j] - k

  let count = 0;
  let prefixCount = new Map<number, number>();
  prefixCount.set(0, 1);
  let prefixSum = 0;

  for (const num of nums) {
    prefixSum += num;
    const need = prefixSum - k;

    if (prefixCount.has(need)) {
      count += prefixCount.get(need)!;
    }

    prefixCount.set(prefixSum, (prefixCount.get(prefixSum) ?? 0) + 1);
  }

  return count;
}

// Time: O(n)
// Space: O(n)
```

### ** [347. Top K Frequent Elements](https://leetcode.com/problems/top-k-frequent-elements/) (2/5)

```typescript
function topKFrequent(nums: number[], k: number): number[] {

    const freq = new Map<number, number>();

    for (const num of nums) {
        freq.set(num, (freq.get(num) ?? 0) + 1);
    }

    const pq = new MaxPriorityQueue({
        compare: (a, b) => {
            if (a[0] !== b[0]) return b[0] - a[0];
            return b[1] - a[1];
        }
    })

    for (const [num, count] of freq) {
        pq.enqueue([count, num]);
    }

    const res: number[] = [];

    for (let i = 0; i < k; i++) {
        res.push(pq.dequeue()[1]);
    }

    return res;
};

// Time: O(nlog(n))
// Space: O(n)



function topKFrequent(nums: number[], k: number): number[] {
  const freq = new Map<number, number>();

  for (const n of nums) {
    freq.set(n, (freq.get(n) ?? 0) + 1);
  }

  const bucket = Array.from({ length: nums.length + 1 }, () => []);

  for (const [num, count] of freq) {
    bucket[count].push(num);
  }

  const res: number[] = [];

  // scan n+1 times in the worst case
  for (let i = bucket.length - 1; i >= 0 && res.length < k; i--) {
    for (const n of bucket[i]) {
      res.push(n);
      if (res.length === k) break;
    }
  }

  return res;
}

// Time: O(n) 
// Space: O(n)
```

```typescript
function groupAnagrams(strs: string[]): string[][] {
  const freq = new Map<string, string[]>();

  for (const str of strs) {
    const sorted = str.split('').sort().join('');

    if (!freq.has(sorted)) {
      freq.set(sorted, []);
    }

    freq.get(sorted).push(str);
  }

  const res: string[][] = [];

  for (const [key, value] of freq) {
    res.push([...value]);
  }

  return res;
}
```

### [3. Longest Substring Without Repeating Characters](https://leetcode.com/problems/longest-substring-without-repeating-characters/) (19/5)

```typescript

function lengthOfLongestSubstring(s: string): number {
  if (!s) return 0;

  const set = new Set<string>();
  let l = 0;
  let maxLen = 0;

  for (let r = 0; r < s.length; r++) {
    const ch = s[r];

    while (set.has(ch)) {
      set.delete(s[l]);
      l++;
    }

    set.add(ch);
    maxLen = Math.max(maxLen, set.size);
  }

  return maxLen;
}


// Time: O(n)
// Space: O(n)
```

### [200. Number of Islands ](https://leetcode.com/problems/number-of-islands/)(20/5)

```typescript

function numIslands(grid: string[][]): number {
  const m = grid.length;
  const n = grid[0].length;

  let count = 0;

  const dfs = (x: number, y: number) => {
    // out of bound
    if (x < 0 || y < 0 || x >= m || y >= n) return;

    if (grid[x][y] !== '1') return;

    grid[x][y] = '0';

    dfs(x - 1, y);
    dfs(x, y + 1);
    dfs(x + 1, y);
    dfs(x, y - 1);
  };

  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (grid[i][j] !== '1') continue;
      dfs(i, j);
      count++;
    }
  }

  return count;
}
```

### **[207. Course Schedule](https://leetcode.com/problems/course-schedule/) (20/5)

```typescript

function canFinish(numCourses: number, prerequisites: number[][]): boolean {
  const preMap = new Map<number, number[]>();

  for (let i = 0; i < numCourses; i++) {
    preMap.set(i, []);
  }

  for (const [cor, pre] of prerequisites) {
    preMap.get(cor).push(pre);
  }

  let cycle = new Set<number>();

  const dfs = (course: number): boolean => {
    if (preMap.get(course).length === 0) return true;
    if (cycle.has(course)) return false;

    cycle.add(course);

    for (let pre of preMap.get(course)) {
      if (!dfs(pre)) return false;
    }

    cycle.delete(course);
    preMap.set(course, []);
    return true;
  };

  for (let i = 0; i < numCourses; i++) {
    if (!dfs(i)) return false;
  }

  return true;
}
```
